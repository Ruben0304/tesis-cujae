# Lista de Símbolos {.unnumbered}

\begin{description}[leftmargin=3.5cm, style=nextline, font=\bfseries]
  \item[$P$]        Potencia eléctrica generada (W)
  \item[$E$]        Energía producida en un intervalo (Wh)
  \item[$G$]        Irradiancia solar global (W/m\textsuperscript{2})
  \item[$A$]        Área efectiva del arreglo fotovoltaico (m\textsuperscript{2})
  \item[$\eta$]     Eficiencia de conversión del módulo (\%)
  \item[$T$]        Temperatura de la celda ($^{\circ}$C)
  \item[$C$]        Capacidad nominal de la batería (kWh)
  \item[$SOC$]      Estado de carga de la batería (\%)
  \item[$\Delta t$] Paso de tiempo de simulación (h)
  \item[$\hat{y}$]  Valor predicho por el modelo
  \item[$y$]        Valor observado o medido
  \item[$\epsilon$] Error de predicción ($y - \hat{y}$)
  \item[$N$]        Número de muestras
\end{description}
